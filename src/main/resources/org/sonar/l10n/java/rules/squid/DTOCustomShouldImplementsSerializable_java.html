com.sinosoft.claim.dto.custom 包下的类必须实现java.io.Serializable 接口。

不合规范代码示例

package com.sinosoft.claim.dto.custom;

import java.io.Serializable;
import java.util.*;
import com.sinosoft.claim.dto.domain.*;


public class AnaResultDto // Noncompliant
{
    private int rowsCount = 0;
    private Collection collection = new ArrayList();
    public AnaResultDto(){
    }
}

合规代码示例

package com.sinosoft.claim.dto.custom;

import java.io.Serializable;
import java.util.*;
import com.sinosoft.claim.dto.domain.*;


public class AnaResultDto implements Serializable // Compliant
{
    private int rowsCount = 0;
    private Collection collection = new ArrayList();
    public AnaResultDto(){
    }
}
