    public PageRecord findByConditions(DBManager dbManager,String conditions,int pageNo,int rowsPerPage) throws Exception{

        DBPrpLcertainLoss dbPrpLcertainLoss = new DBPrpLcertainLoss(dbManager);

        collection = dbPrpLcertainLoss.findByQueryCondition(conditions,pageNo,rowsPerPage);

        PageRecord pageRecord2 = new PageRecord(count,pageNo,1,rowsPerPage,collection);// Noncompliant
