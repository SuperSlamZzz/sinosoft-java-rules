在查询和翻页查询的方法中，必须使用 request.setAttribute("fm", new AbstractForm(pageRecord));把PageRecord 对象存放到 javax.servlet.http.HttpServletRequest 对象中。

不合规范代码示例

package com.sinosoft.claim.ui.control.facade;

public class UIBlackListFacade extends Action{
    public ActionForward execute(ActionMapping mapping, HttpServletRequest request)
            throws Exception {
        PageRecord pageRecord = (PageRecord)blUwblacklistFacade.findByConditionsAndPrpCmain(conditions, intPageNo, intRecordPerPage);
        Collection collection = pageRecord.getResult();
        request.setAttribute("abstractForm", new AbstractForm(collection)); // Noncomplaint
}

合规代码示例

package com.sinosoft.claim.ui.control.facade;

public class UIBlackListFacade extends Action{
    public ActionForward execute(ActionMapping mapping, HttpServletRequest request)
            throws Exception {
        PageRecord pageRecord = (PageRecord)blUwblacklistFacade.findByConditionsAndPrpCmain(conditions, intPageNo, intRecordPerPage);
        Collection collection = pageRecord.getResult();
        request.setAttribute("abstractForm", new AbstractForm(pageRecord)); // Complaint
}
